% Define document class and default bibliography style
\documentclass[conference]{IEEEtran}
\newcommand{\paperbibliographystyle}{plain}

% Pre-abstract formatting
\newcommand{\preabstract}{
    \title{\papertitle}
    
    % author block
    \author{
        \ifcsname doubleblind\endcsname
            \IEEEauthorblockN{\doubleblind}
        \else
            \ifcsname firstauthor\endcsname
                \IEEEauthorblockN{\firstauthor}
                \IEEEauthorblockA{\institution{\firstauthoraffiliation}}
                \and
            \fi
            \ifcsname collaboratorone\endcsname
                \IEEEauthorblockN{\collaboratorone}
                \IEEEauthorblockA{\institution{\collaboratoroneaffiliation}}
                \and
            \fi
            \ifcsname collaboratortwo\endcsname
                \IEEEauthorblockN{\collaboratortwo}
                \IEEEauthorblockA{\institution{\collaboratortwoaffiliation}}
                \and
            \fi
            \ifcsname collaboratorthree\endcsname
                \IEEEauthorblockN{\collaboratorthree}
                \IEEEauthorblockA{\institution{\collaboratorthreeaffiliation}}
                \and
            \fi
            \IEEEauthorblockN{\lastauthor}
            \IEEEauthorblockA{\institution{\lastauthoraffiliation}}
        \fi
    }
    \maketitle
}
